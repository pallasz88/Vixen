206template <Colors sideToMove> constexpr void MoveGenerator::GenerateCastlingMoves(const Board &board) noexcept
207{
208 const auto currentCastlingRights = board.GetCastlingRights();
209 const auto emptySquaresBitBoard = board.GetBitBoard(' ');
210
211 const std::array castleEmptySquareLookUp = {IsBitSet(emptySquaresBitBoard, Squares::B8, Squares::C8, Squares::D8),
212 IsBitSet(emptySquaresBitBoard, Squares::F8, Squares::G8),
213 IsBitSet(emptySquaresBitBoard, Squares::B1, Squares::C1, Squares::D1),
214 IsBitSet(emptySquaresBitBoard, Squares::F1, Squares::G1)};
215
216 const auto isCastlingAvailable = [&board, currentCastlingRights,
217 &castleEmptySquareLookUp](CastlingRights castlingRight) {
218 const unsigned castlingRightBB = static_cast<uint8_t>(castlingRight);
219 return (currentCastlingRights & castlingRightBB) &&
220 castleEmptySquareLookUp[Constants::castlingLookUp[castlingRight]] &&
221 !Check::IsInCheck<sideToMove>(board) &&
222 !Check::IsSquareAttacked<sideToMove>(Constants::rookSquareAfterCastling[castlingRight], board);
223 };
224
225 if constexpr (sideToMove == Colors::WHITE)
226 {
227 if (isCastlingAvailable(CastlingRights::WKCA))
228 moveList.emplace_back(static_cast<unsigned>(Squares::E1), static_cast<unsigned>(Squares::G1),
229 static_cast<uint8_t>(MoveTypes::KING_CASTLE));
230
231 if (isCastlingAvailable(CastlingRights::WQCA))
232 moveList.emplace_back(static_cast<unsigned>(Squares::E1), static_cast<unsigned>(Squares::C1),
233 static_cast<uint8_t>(MoveTypes::QUEEN_CASTLE));
234 }
235 else
236 {
237 if (isCastlingAvailable(CastlingRights::BKCA))
238 moveList.emplace_back(static_cast<unsigned>(Squares::E8), static_cast<unsigned>(Squares::G8),
239 static_cast<uint8_t>(MoveTypes::KING_CASTLE));
240
241 if (isCastlingAvailable(CastlingRights::BQCA))
