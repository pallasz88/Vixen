31template <size_t N, char delimiter>
32[[nodiscard]] constexpr auto Board::SplitFenPosition(std::string_view position) const noexcept
33{
34 std::array<std::string_view, N> parts;
35 auto part = 0ULL;
36 auto firstPosition = 0ULL;
37 auto nextPosition = position.find_first_of(delimiter);
38 while (part < N)
39 {
40 parts[part++] = position.substr(firstPosition, nextPosition - firstPosition);
41 firstPosition = nextPosition + 1;
42 nextPosition = position.find_first_of(delimiter, firstPosition);
43 }
44 return parts;
45}

95constexpr void Board::ParseFenPiecePart(std::string_view parsedPosition)
96{
97 unsigned squareIndex = Constants::MAX_SQUARE_INDEX;
98 for (const auto &fenChar : parsedPosition)
99 {
100 switch (fenChar)
101 {
102 case 'P':
103 case 'N':
104 case 'B':
105 case 'R':
106 case 'Q':
107 case 'K':
108 case 'p':
109 case 'n':
110 case 'b':
111 case 'r':
112 case 'q':
113 case 'k':
114 SetBit(bitBoards[static_cast<unsigned>(GetPieceIndex(static_cast<uint8_t>(fenChar)))], squareIndex);
115 material += GetPieceMaterial(fenChar);
116 pieceList[squareIndex] = fenChar;
117 break;
118 case '/':
119 continue;
120 case '1':
121 case '2':
122 case '3':
123 case '4':
124 case '5':
125 case '6':
126 case '7':
127 case '8':
128 squareIndex -= static_cast<unsigned>(fenChar - '1');
129 break;
130 default:
131 std::cerr << "ERROR IN FEN: PIECE POSITION"
132 << "\n";
133 return;
134 }
135 --squareIndex;
136 }
137}

140{
141 bitBoards[GetPieceIndex('F')] = bitBoards[GetPieceIndex('K')] | bitBoards[GetPieceIndex('Q')] |
142 bitBoards[GetPieceIndex('R')] | bitBoards[GetPieceIndex('B')] |
143 bitBoards[GetPieceIndex('N')] | bitBoards[GetPieceIndex('P')];

186bool Board::MakeMove(vixen::Move move)
187{
188 const auto from = move.GetFromSquare();
189 const auto to = move.GetToSquare();
190 const auto moveType = move.GetMoveType();
191 const auto enPassantSquare = whiteToMove ? to - 8 : to + 8;
192 const char movingPieceLetter = pieceList[from];
193 const char capturedPieceLetter = pieceList[to];
194
195 history[historyPly++] = History{enPassantBitBoard, hashBoard.GetHash(), castlingRights, fiftyMoves, move,
196 movingPieceLetter, capturedPieceLetter};
197
198 ++fiftyMoves;
199
200 RemovePiece(from, movingPieceLetter);
201 UpdateCastlingRights(from, to);
202
203 if (enPassantBitBoard != Constants::EMPTY_BOARD)
204 {
205 hashBoard.HashEnPassant(enPassantBitBoard);
206 enPassantBitBoard = Constants::EMPTY_BOARD;
207 }
208
209 if (IsMovingPawn(movingPieceLetter))
210 fiftyMoves = 0;
211
212 if (moveType & static_cast<uint8_t>(MoveTypes::CAPTURE))
213 MakeCapture(to, capturedPieceLetter, moveType);
214
215 else if (moveType == static_cast<uint8_t>(MoveTypes::DOUBLE_PAWN_PUSH))
216 MakeDoublePawnPush(enPassantSquare);
217
218 else if (moveType == static_cast<uint8_t>(MoveTypes::KING_CASTLE))
219 whiteToMove ? MoveCastlingWhiteRook(static_cast<unsigned>(Squares::H1), static_cast<unsigned>(Squares::F1))
220 : MoveCastlingBlackRook(static_cast<unsigned>(Squares::H8), static_cast<unsigned>(Squares::F8));
221
222 else if (moveType == static_cast<uint8_t>(MoveTypes::QUEEN_CASTLE))
223 whiteToMove ? MoveCastlingWhiteRook(static_cast<unsigned>(Squares::A1), static_cast<unsigned>(Squares::D1))
224 : MoveCastlingBlackRook(static_cast<unsigned>(Squares::A8), static_cast<unsigned>(Squares::D8));
225
226 if (moveType & static_cast<uint8_t>(MoveTypes::PROMOTION))
227 {
228 char promotion = whiteToMove ? "NBRQ"[moveType & 3U] : "nbrq"[moveType & 3U];
229 material += GetPieceMaterial(promotion) - GetPieceMaterial(movingPieceLetter);
230 AddPiece(to, promotion);
231 }
232 else
233 AddPiece(to, movingPieceLetter);
234
235 whiteToMove = !whiteToMove;
236 hashBoard.HashSide();
237
238 if (whiteToMove ? Check::IsInCheck<Colors::BLACK>(*this) : Check::IsInCheck<Colors::WHITE>(*this))
239 {
240 TakeBack();
241 return false;
242 }
243 return true;
244}

287void Board::TakeBack()
288{
289 if (historyPly == 0)
290 throw std::runtime_error("Empty history");
291
292 const History &lastPosition = history[--historyPly];
293
294 const auto from = lastPosition.move.GetFromSquare();
295 const auto to = lastPosition.move.GetToSquare();
296 const auto moveType = lastPosition.move.GetMoveType();
297 const char movingPieceLetter = lastPosition.movedPiece;
298 const char capturedPieceLetter = lastPosition.capturedPiece;
299
300 whiteToMove = !whiteToMove;
301 fiftyMoves = lastPosition.fiftyMoves;
302 enPassantBitBoard = lastPosition.enPassant;
303 castlingRights = lastPosition.castlingRights;
304
305 if (moveType == static_cast<uint8_t>(MoveTypes::KING_CASTLE))
306 whiteToMove ? MoveCastlingWhiteRook(static_cast<unsigned>(Squares::F1), static_cast<unsigned>(Squares::H1))
307 : MoveCastlingBlackRook(static_cast<unsigned>(Squares::F8), static_cast<unsigned>(Squares::H8));
308
309 else if (moveType == static_cast<uint8_t>(MoveTypes::QUEEN_CASTLE))
310 whiteToMove ? MoveCastlingWhiteRook(static_cast<unsigned>(Squares::D1), static_cast<unsigned>(Squares::A1))
311 : MoveCastlingBlackRook(static_cast<unsigned>(Squares::D8), static_cast<unsigned>(Squares::A8));
312
313 else if (moveType & static_cast<uint8_t>(MoveTypes::PROMOTION))
314 {
315 char promotion = whiteToMove ? "NBRQ"[moveType & 3U] : "nbrq"[moveType & 3U];
316 material -= GetPieceMaterial(promotion) - GetPieceMaterial(movingPieceLetter);
317 RemovePiece(to, promotion);
318 AddPiece(to, movingPieceLetter);
319 }
320
321 RemovePiece(to, movingPieceLetter);
322 AddPiece(from, movingPieceLetter);
323
324 if (moveType == static_cast<uint8_t>(MoveTypes::ENPASSANT))
325 {
326 whiteToMove ? AddPiece(to - 8, 'p') : AddPiece(to + 8, 'P');
327 material += whiteToMove ? GetPieceMaterial('p') : GetPieceMaterial('P');
328 }
329
330 else if (moveType & static_cast<uint8_t>(MoveTypes::CAPTURE))
331 {
332 material += GetPieceMaterial(capturedPieceLetter);
333 AddPiece(to, capturedPieceLetter);
334 }
335
336 hashBoard.SetHash(lastPosition.hash);
337}

365bool Board::MakeMove(std::string_view notation)
366{
367 const std::regex moveRegex{"[a-h][1-8][a-h][1-8][q|r|b|n]?"};
368
369 if (std::match_results<std::string_view::const_iterator> iterator;
370 !std::regex_match(begin(notation), end(notation), iterator, moveRegex))
371 throw std::runtime_error("INVALID MOVE FORMAT. Use for example e2e4.");
372
373 const auto from = static_cast<unsigned>(Move::NotationToSquare(notation.substr(0, 2)));
374 const auto to = static_cast<unsigned>(Move::NotationToSquare(notation.substr(2, 2)));
375
376 char promoted = ' ';
377 if (notation.size() == 5)
378 promoted = notation.back();
379
380 uint8_t moveType = 0;
381 if (promoted != ' ')
382 {
383 moveType = static_cast<uint8_t>(GetPromotionType(promoted));
384 const char capturedPiece = GetPieceList()[to];
385 if (capturedPiece != ' ')
386 moveType |= static_cast<uint8_t>(MoveTypes::CAPTURE);
387 }
388
389 const auto decodedPromotion = Move(from, to, moveType);
390 const auto decodedMove = decodedPromotion.RemoveMoveType();
391
392 for (const auto &move : GetMoveList<MoveTypes::ALL_MOVE>())
393 if ((move.RemoveMoveType()) == decodedMove)
394 return (moveType == 0) ? MakeMove(move) : MakeMove(decodedPromotion);
395
396 return false;
397}
